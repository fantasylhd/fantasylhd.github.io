1.概述

LinkedkList是基于节点实现的双向链表，当前的每一个节点都是指向前一个节点和后一个节点从而实现链表

4.1 Node

private static class Node<E> {
    //存储数据的属性
    E item;
    //存储上一节点的地址
    Node<E> next;
    //存储下一节点的地址
    Node<E> prev;

    Node(Node<E> prev, E element, Node<E> next) {
        this.item = element;
        this.next = next;
        this.prev = prev;
    }
}

5.1 添加单个元素 - add()

public boolean add(E e) {
    // 调用linkLast方法
    linkLast(e);
    return true;
}

void linkLast(E e) {
     // 创建一个l指针指向last指针
     final Node<E> l = last;
     // 创建一个新节点 prev指针指向当前链表的最后一个节点  next指针指向null
     final Node<E> newNode = new Node<>(l, e, null);
     // last指向新节点
     last = newNode;
     //如果l指向空 说明创建的是第一个节点
     if (l == null)
       //first指向新节点
       first = newNode;
     else
       //l的next指向新节点
       l.next = newNode;
       //大小++
       size++;
       //修改次数+
       modCount++;
 }

元素的添加的示意图

说明：定义的这个指针l总是会指向最后一个节点,如果是第一个节点则指向null

思考：1.相比ArrayList无需考虑扩容问题 2. 添加的时间复杂度为0(1)

5.2 添加元素到指定位置() - add(int index, E element)

public void add(int index, E element) {
    //检查下标是否合法
    checkPositionIndex(index);
    //如果是要插入到最后一个元素后面
    if (index == size)
        //链接到最后一个节点
        linkLast(element);
    else
        //先找到下标为位置节点,链接到节点之前
        linkBefore(element, node(index));
}

Node<E> node(int index) {
        //如果下标小于size/2，则从前开始找
    if (index < (size >> 1)) {
         Node<E> x = first;
         for (int i = 0; i < index; i++)
            x = x.next;
           return x;
     } else {  //大于size/2则从后开始找
            Node<E> x = last;
         for (int i = size - 1; i > index; i--)
             x = x.prev;
            return x
     }
}

void linkBefore(E e, Node<E> succ) {
      //下标位置节点的前一个节点
      final Node<E> pred = succ.prev;
      //新节点 prev指针指向前一个节点 next指向下标位置节点
      final Node<E> newNode = new Node<>(pred, e, succ);
      //修改下标节点的prev指针
      succ.prev = newNode;
      //如果pred是null 说明链表只有一个节点
      if (pred == null)
         //first指针指向新节点
         first = newNode;  
      else
         //前一个节点的next指针指向新节点
         pred.next = newNode;
      size++;
      modCount++;
}

说明: 插入节点需要先找到前一个节点，使用双向链表能减少一半的时间复杂度

思考：牺牲少部分的内存空间换取时间复杂度

5.3 添加多个元素 - (addAll(Collection<? extends E> c))

public boolean addAll(Collection<? extends E> c) {
    //调用添加到指定位置多个元素的方法
    return addAll(size, c);
}

public boolean addAll(int index, Collection<? extends E> c) {
    checkPositionIndex(index);
    //转化成数组
    Object[] a = c.toArray();
    //新数量
    int numNew = a.length;
    //转化之后数组长度为0 返回失败
    if (numNew == 0)
        return false;
    //定义前一个节点和下标位置节点的指针
    Node<E> pred, succ;
    //如果下标 == 链表的大小  说明：链表要么为空并且插入下标为0 || 要么是要插入到链表最后一个节点的后面
    if (index == size) {
        //下标位置节点为null
        succ = null;
        //pred指针应该指向最后一个节点
        pred = last;
    } else {
        //找到下标位置节点
        succ = node(index);
        // 前面一个节点 = 下标位置节点的prev指针指向的节点
        pred = succ.prev;
    }

    //遍历转化后的数组
    for (Object o : a) {  
        //进行泛型类型的转换
        @SuppressWarnings("unchecked") E e = (E) o; 
        //创建新节点，prev指针指向前一个节点 next指针指向null
        Node<E> newNode = new Node<>(pred, e, null);
        // 如果前面一个节点 == null 说明原链表中可能:  1.只有一个节点  || 2.没有节点 || 3. 添加到0位置
        if (pred == null)
            //first指向新节点
            first = newNode;
        else
            //前一个节点next指向新节点
            pred.next = newNode;
        //下一个循环的前一个节点就是这轮循环的新节点
        pred = newNode;
    }

    // 当前下标节点为空  说明：链表要么为空 要么是要插入到链表最后一个节点的后面
    if (succ == null) {
        //last指针指向数组元素的最后一个节点
        last = pred;
    } else {
        //数组最后一个节点指向下标位置节点
        pred.next = succ;
        //下标位置节点的前一个节点就是数组的最后一个节点
        succ.prev = pred;
    }

    //大小为数组的长度
    size += numNew;
    modCount++;
    return true;
}

5.4 移除单个元素 - remove(int index)

public E remove(int index) {
    //检查下标是否合法  index>=0 && index < size
    checkElementIndex(index);
           //调用方法进行移除
    return unlink(node(index));
}

E unlink(Node<E> x) {
    //保存删除节点的值 已便于返回
    final E element = x.item;
    //删除节点的前一个节点
    final Node<E> next = x.next;
    //删除节点的后一个节点
    final Node<E> prev = x.prev;
    //如果要删除的节点的前一个节点为null 说明: 只存在一个节点 || 要删除的是第一个节点
    if (prev == null) {
        //first应指向后一个节点
        first = next;  
    } else {
        //前一个节点跳过要删除的节点
        prev.next = next;
        //断开要删除节点的前指向
        x.prev = null;
    }
    //如果要删除节点的后一个节点为null 说明: 只存在一个节点 || 要删除的是最后一个节点
    if (next == null) {
        //last应指向前一个节点
        last = prev;
    } else {
        //后一个节点的前指向应指向前节点
        next.prev = prev;
        //断开要删除节点的后指向
        x.next = null;
    }
    //将删除节点的元素值置为null
    x.item = null;
    //大小--
    size--;
    modCount++;
    return element;

}

5.5 移除单个元素方式2 - remove(Object o)

public boolean remove(Object o) {
    //可以删除值为null的节点
    if (o == null) { 
        for (Node<E> x = first; x != null; x = x.next) {
            //先找出为null的节点
            if (x.item == null) {
                //调用方法进行删除
                unlink(x);
                return true;
            }
        }
    } else { //下面同理
        for (Node<E> x = first; x != null; x = x.next) {
            if (o.equals(x.item)) {
                unlink(x);
                return true;
            }
        }
    }
    return false;
}

说明: 虽然移除操作的时间复杂度是O(1)的，但是得先找到要删除节点，遍历所花得时间为O(n)的时间复杂度

5.6 移除多个元素 - removeAll(Collection<?> c)

public boolean removeAll(Collection<?> c) {
    //判断是否为空
    Objects.requireNonNull(c);
    //是否删除了元素的标记
    boolean modified = false;
    //获取迭代器
    Iterator<?> it = iterator();
    while (it.hasNext()) {
        //如果包含这个元素
        if (c.contains(it.next())) {
            //调用迭代器的删除方法进行删除
            it.remove();
            //修改删除标记
            modified = true;
        }
    }
    return modified;
}

5.7 查找单个元素位置 - indexOf(Object o)

public int indexOf(Object o) {
    //从0开始找
    int index = 0;
    //为空的情况
    if (o == null) {
        for (Node<E> x = first; x != null; x = x.next) {
            if (x.item == null)
                return index;
            index++;
        }
    } else {
        for (Node<E> x = first; x != null; x = x.next) {
            if (o.equals(x.item))
                return index;
            index++;
        }
    }
    return -1;
}

5.8 获得指定位置的值 - get(int index)

public E get(int index) {
    //检查下标是否合法
    checkElementIndex(index);
    //借助node方法寻找节点返回值
    return node(index).item;
}

5.9 设置指定位置的值 - set(int index, E element)

public E set(int index, E element) {
    checkElementIndex(index);
    Node<E> x = node(index);
    E oldVal = x.item;
    x.item = element;
    return oldVal;
}
//借助node方法找到节点，返回旧值，将新值设置进去即可

最终呢是调用 LinkedList 对 listIterator(int index)的实现

5.11 创建 ListIterator 迭代器 - listIterator(int index)

public ListIterator<E> listIterator(int index) {
    checkPositionIndex(index);
    return new ListItr(index);
}

private class ListItr implements ListIterator<E> {
    //上一次访问的节点
    private Node<E> lastReturned;
    //下一个访问节点
    private Node<E> next;
    //下一个访问的索引
    private int nextIndex;
    //期望的版本号
    private int expectedModCount = modCount;

    ListItr(int index) {
        //初始化next的值 传入的值是index == size 说明想从后往前遍历，否则是从前往后遍历
        next = (index == size) ? null : node(index);
        //初始化访问索引
        nextIndex = index;  
    }
    //用来判断是否还有节点的方法  ----  从前往后
    public boolean hasNext() {
        //如果访问下一索引<size 说明还有节点
        return nextIndex < size;
    }
    //用来访问节点的方法
    public E next() {
        //检查版本号是否有问题
        checkForComodification();
        //判断是否有节点
        if (!hasNext())
            throw new NoSuchElementException();

        //初始化上一次访问节点
        lastReturned = next;
        //移动指针到下一节点
        next = next.next;
        //访问索引++
        nextIndex++;
        //返回本次访问的值
        return lastReturned.item;
    }
    //用来判断是否还有节点的方法  ----- 从后往前
    public boolean hasPrevious() {
        //访问的下一索引 > 0 说明还有元素
        return nextIndex > 0;
    }

    //用来访问节点的方法  ----- 从后往前
    public E previous() {
        checkForComodification();
        //判断是否有节点
        if (!hasPrevious())
            throw new NoSuchElementException();
        //初始化本次访问节点
        lastReturned = next = (next == null) ? last : next.prev;
        //访问索引--
        nextIndex--;
        //返回节点值
        return lastReturned.item;
    }

    //返回下次访问的索引   ----- 从前往后
    public int nextIndex() {
        return nextIndex;
    }

    //返回下次访问的索引   ----- 从后往前
    public int previousIndex() {
        return nextIndex - 1;
    }

    //用来在迭代器中进行删除的方法
    public void remove() {
        checkForComodification();
        //上次访问节点为null
        if (lastReturned == null)
            throw new IllegalStateException();
        //上次访问节点的下一节点
        Node<E> lastNext = lastReturned.next;
        //删除节点
        unlink(lastReturned);
        //此处是因为调用的是从后往前遍历的previous()方法才能成立：  
        //因为是从后面进行删除：LastNext为null next也要赋值为null，方便下次的previous()取到last即是最后一个节点
        if (next == lastReturned)
            next = lastNext;
        else
            //删除节点后 下一个访问的节点下标应该--
            nextIndex--;
        lastReturned = null;
        expectedModCount++;
    }
    //用来在迭代器中设置值的方法
    public void set(E e) {
        if (lastReturned == null)
            throw new IllegalStateException();
        checkForComodification();
        //将节点值置为新值即可
        lastReturned.item = e;
    }
    //用来在迭代器中进行添加的方法
    public void add(E e) {
        checkForComodification();
        lastReturned = null;
        //如果不存在下一节点
        if (next == null)
            //就是链接到最后一个节点后
            linkLast(e);
        else
            //往下一节点之前链接节点
            linkBefore(e, next);
        //新增了一个节点 应该下一个访问的索引应该++
        nextIndex++;
        expectedModCount++;
    }

    //还不情况是干什么的
    public void forEachRemaining(Consumer<? super E> action) {
        Objects.requireNonNull(action);
        while (modCount == expectedModCount && nextIndex < size) {
            action.accept(next.item);
            lastReturned = next;
            next = next.next;
            nextIndex++;
        }
        checkForComodification();
    }
    //检查版本号
    final void checkForComodification() {
        if (modCount != expectedModCount)
            throw new ConcurrentModificationException();
    }

}

说明: 因为LinkedList要实现双向的访问迭代访问，所以使用iterator()接口就不行了，因为iterator()只支持从前往后访问，所以Java新增了

listIterator接口来支持向前和向后的迭代方法

6. 总结

1.LinkedList还提供队列、栈、双端队列的功能，因为双向链表比较适合来实现这些功能

2.LinkedList相比于ArrayList来说添加元素和删除元素到指定的位置的效率要高
